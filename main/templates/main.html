Objects

User defined variables that have their owm functions
Objects have special methods called dunder methods ex: __init__

class Coordinate: \n \t def __init__(self, x=0, y=0): \n \t \t self.xpos = x \n \t \t self.ypos = y \n \n \t def __eq__(self, other): \n \t \t if not isinstance(other, Coordinate): \n \t \t \t return False \n \t \t return other.xpos == self.xpos and other.ypos == self.ypos \n

# You can have more than one class in a file \n class Person: \n \t # Constructor \n \t def __init__(self, name, age): \n \t \t self.name = name \n \t \t self.age = age \n \t # Printing as string \n \t def __repr__(self): \n \t \t msg0 = " Name: " + self.name \n \t \t msg1 = " Age: {}" \n \t \t msg1 = msg1.format(self.age) \n \t \t return msg0 + msg1 \n \n # Inheritance \n class Student(Person): \n \t def __init__(self, name, age, classnum): \n \t \t super().__init__(name, age) \n \t \t self.classnum = classnum \n \t def __repr__(self): \n \t \t msg0 = Person.__str__(self) \n \t \t msg1 = " Class: {}" \n \t \t msg1 = msg1.format(self.classnum) \n \t \t return msg0 + msg1 \n \n class Employee: \n \t def __init__(self, position, salary): \n \t \t self.position = position \n \t \t self.salary = salary \n \t def __repr__(self): \n \t \t msg0 = " Position:" + self.position \n \t \t msg1 = " Salary: {}" \n \t \t msg1 = msg1.format(self.salary) \n \t \t return msg0 + msg1 \n \n # Multiple Inheritance \n class Teacher(Person, Employee): \n \t def __init__(self, name, age, salary): \n \t \t Person.__init__(self, name, age) \n \t \t Employee.__init__(self, "Teacher", salary) \n \t def __repr__(self): \n \t \t msg0 = Person.__repr__(self) \n \t \t msg1 = Employee.__repr__(self) \n \t \t return msg0 + msg1 \n
